const form = document.getElementById('applicationForm'); const successMsg = document.getElementById('successMessage'); // === ВСТАВЬТЕ ВАШИ ДАННЫЕ === const WEBHOOK_URL = 'https://discord.com/api/webhooks/1517962629002301612/4X4QoaCN84kacOXgkew8suY9BCwbpKkDEIXJHh7O7z-lgkn45eNjhq2F14atjsbcSeYm'; // URL вебхука из Discord const FORMSPREE_URL = 'https://formspree.io/f/mojoaqee'; // Ваш новый ID формы form.addEventListener('submit', async (e) => { e.preventDefault(); // Собираем данные из формы const formData = new FormData(form); const applicationId = Date.now().toString(36) + Math.random().toString(36).substr(2, 5); const applicationData = { id: applicationId, name: formData.get('name'), age: formData.get('age'), role: formData.get('role'), discord: formData.get('discord'), experience: formData.get('experience') || 'Не указан', rules: formData.get('rules') ? '✅ Да' : '❌ Нет', responsibility: formData.get('responsibility') ? '✅ Да' : '❌ Нет', timestamp: new Date().toLocaleString('ru-RU') }; // === 1. ОТПРАВКА В DISCORD (с кнопками) === const discordEmbed = { title: '📝 Новая заявка!', color: 0x8B5CF6, fields: [ { name: '🆔 ID заявки', value: applicationData.id, inline: false }, { name: '👤 Имя', value: applicationData.name, inline: true }, { name: '📅 Возраст', value: applicationData.age, inline: true }, { name: '🎭 Роль', value: applicationData.role, inline: false }, { name: '💬 Discord', value: applicationData.discord, inline: false }, { name: '📖 Опыт', value: applicationData.experience, inline: false }, { name: '✅ Согласие с правилами', value: applicationData.rules, inline: true }, { name: '⚖️ Ответственность', value: applicationData.responsibility, inline: true }, { name: '🕐 Время', value: applicationData.timestamp, inline: false } ], footer: { text: 'Lost Signal Studio' } }; const discordPayload = { embeds: [discordEmbed], components: [ { type: 1, components: [ { type: 2, label: '✅ Принять', style: 3, custom_id: `accept_${applicationData.id}` }, { type: 2, label: '❌ Отклонить', style: 4, custom_id: `decline_${applicationData.id}` } ] } ] }; // === 2. ОТПРАВКА В FORMSPREE (на почту) === const formspreeData = new FormData(form); // Добавляем дополнительное поле с ID заявки formspreeData.append('application_id', applicationData.id); try { // Отправляем в Discord const discordResponse = await fetch(WEBHOOK_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(discordPayload) }); // Отправляем в Formspree const formspreeResponse = await fetch(FORMSPREE_URL, { method: 'POST', body: formspreeData, headers: { 'Accept': 'application/json' } }); // Проверяем обе отправки if (discordResponse.ok && formspreeResponse.ok) { // Сохраняем заявку в локальном хранилище (опционально) const saved = JSON.parse(localStorage.getItem('applications') || '[]'); saved.push(applicationData); localStorage.setItem('applications', JSON.stringify(saved)); // Показываем сообщение об успехе form.style.display = 'none'; successMsg.style.display = 'block'; successMsg.scrollIntoView({ behavior: 'smooth' }); } else { alert('❌ Не удалось отправить заявку. Попробуйте ещё раз.'); } } catch (error) { alert('❌ Ошибка соединения. Проверьте интернет и попробуйте снова.'); console.error('Ошибка:', error); } });
